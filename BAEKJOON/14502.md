```python
import sys
input = sys.stdin.readline
import copy

def find(cnt):
    global result
    if cnt == 3: # 벽 3개 채웠다면
        virus_arr = copy.deepcopy(arr) # 바이러스 퍼뜨리기 위해 복사
        for r in range(N):
            for c in range(M):
                if virus_arr[r][c] == 2:
                    virus(r, c, virus_arr)
        count = 0 # 현재 안전지대 개수
        for i in range(N):
            count += virus_arr[i].count(0)
        result = max(result, count) # 최대 안전지대 개수
        return
    # else: # 만약 벽의 개수가 3개가 아니라면 한 군데씩 넣기
    for i in range(N):
        for j in range(M):
            if arr[i][j] == 0:
                arr[i][j] = 1
                cnt +=  1
                find(cnt)
                arr[i][j] = 0
                cnt -= 1

def virus(x, y, virus_arr):
    for i in range(4):
        nr, nc = x + dr[i], y + dc[i]
        if 0 <= nr < N and 0 <= nc < M and virus_arr[nr][nc] == 0: # 바이러스 감염 가능 지역이면 감염
            virus_arr[nr][nc] = 2
            virus(nr, nc, virus_arr)

N, M = map(int, input().split())
arr = [list(map(int, input().split())) for _ in range(N)]
dr = [-1, 0, 1, 0]
dc = [0, 1, 0, -1]
result = 0
find(0) # 벽 만들기
print(result)
```

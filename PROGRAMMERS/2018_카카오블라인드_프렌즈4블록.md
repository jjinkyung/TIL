```python
def binary(arr, n):
    bin_arr = []
    for i in arr:
        bin = []
        while i >= 1:
            bin.append(i%2)
            i = i//2
        if len(bin) < n:
            for _ in range(n-len(bin)):
                bin.append(0)
        bin_arr.append(bin)
    return bin_arr


def solution(n, arr1, arr2):
    answer = []
    bin1 = binary(arr1, n)
    bin2 = binary(arr2, n)
    for i in range(n):
        tmp = ''
        for _ in range(n):
            one = bin1[i].pop(-1)
            two = bin2[i].pop(-1)
            if one == 1 or two == 1:
                tmp += '#'
            elif one == 0 and two == 0:
                tmp += ' '
        answer.append(tmp)
    return answer
```
